//반복문 loop

//for문 (명확한 횟수가 정해져있을 때 사용하면 좋음)

for(let i = 0; i < 10; i++){
  console.log(i+1)
}
_______________________________________________________________________________
//while문 (그게 아니라면 while문 사용)

let i = 0;
while(i<10){
console.log(i+1)
  i++;
}
_______________________________________________________________________________
//do..while문

let i = 0;

do {
  i++;
}
while(i<10)
_______________________________________________________________________________
//break, continue

break: 멈추고 빠져나옴
continue: 멈추고 다음 반복으로 진행
_______________________________________________________________________________

  let answer = confirm('계속할까요?');
